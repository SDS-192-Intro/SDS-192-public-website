hospitals_ma <-
  hospitals %>%
  select(Provider.CCN, Hospital.Name, City, State.Code, Fiscal.Year.End.Date, CCN.Facility.Type, Rural.Versus.Urban, Number.of.Beds) %>%
  filter(State.Code == "MA")
for (i in 1:nrow(hospitals_ma))
{
  print(paste("Provider.CCN:", hospitals$Provider.CCN[i]))
  print(paste("Hospital Name:", hospitals$Hospital.Name[i]))
  print(paste("City:", hospitals$City[i]))
  print(paste("Fiscal.Year.End.Date:", hospitals$Fiscal.Year.End.Date[i]))
  print(paste("CCN.Facility.Type:", hospitals$CCN.Facility.Type[i]))
  print(paste("Rural.Versus.Urban:", hospitals$Rural.Versus.Urban[i]))
  print(paste("Number.of.Beds:", hospitals$Number.of.Beds[i]))
  cat("\\newpage")
}
